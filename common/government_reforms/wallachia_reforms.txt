state_appropriation_of_the_robie = {
	icon = "state_appropriation_of_the_robie"
	allow_normal_conversion = yes
	potential = {
		culture_group = romanian_culture_group
	}
	trigger = {
		custom_trigger_tooltip = {
			tooltip = has_not_abolished_slavery_tooltip
			NOT = {
				has_country_modifier = the_abolish_slavery_act
			}
		}
	}
	modifiers = {
		production_efficiency = 0.15
		build_cost = -0.1
		development_cost = -0.1
		all_estate_loyalty_equilibrium = -0.1
		stability_cost_modifier = 0.1
	}
	effect = {
		if = {
			limit = {
				NOT = {
					has_country_flag = hide_state_appropriation_of_the_robie_tooltip
				}
			}
			if = {
				limit = {
					has_idea_group = humanist_ideas
				}
				custom_tooltip = state_appropriation_of_the_robie_warning_tooltip
				hidden_effect = {
					country_event = {
						id = AB_wallachia_events.11
						days = 1
					}
				}
			}
			set_country_flag = hide_state_appropriation_of_the_robie_tooltip
			add_estate_loyalty = {
				estate = all
				loyalty = -20
			}
			if = {
				limit = {
					NOT = {
						has_country_modifier = wallachian_terror_purging_the_estate_elites
					}
				}
				if = {
					limit = {
						has_estate = estate_nobles
					}
					random_owned_province = {
						spawn_rebels = {
							size = 1
							type = noble_rebels
						}
					}
				}
				if = {
					limit = {
						has_estate = estate_church
					}
					random_owned_province = {
						spawn_rebels = {
							size = 1
							type = orthodox_rebels
						}
					}
				}
				if = {
					limit = {
						has_estate = estate_burghers
					}
					random_owned_province = {
						spawn_rebels = {
							size = 1
							type = particularist_rebels
						}
					}
				}
				if = {
					limit = {
						has_estate = estate_cossacks
					}
					random_owned_province = {
						spawn_rebels = {
							size = 1
							type = cossack_rebels
						}
					}
				}
			}
		}
	}
	removed_effect = {
		clr_country_flag = hide_state_appropriation_of_the_robie_tooltip
	}
	custom_attributes = {
		blocks_humanist_ideas = yes
	}
