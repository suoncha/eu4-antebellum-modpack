#Checks if targetA has more prestige than targetB
has_more_prestige_than = {
	variable_arithmetic_trigger = {
		[[tooltip]
		custom_tooltip = $tooltip$
		]
		export_to_variable = {
			which = target_a_prestige
			value = prestige
			who = $targetA$
		}
		export_to_variable = {
			which = target_b_prestige
			value = prestige
			who = $targetB$
		}
		NOT = {
			check_variable = {
				which = target_b_prestige
				which = target_a_prestige
			}
		}
	}
}
