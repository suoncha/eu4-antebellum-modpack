#Cossack Culture Spread (UKR/Russ/HET) Spread 
province_event = {
	id = ab_new_cultures.23
	title = ab_new_cultures.23.t
	desc = ab_new_cultures.23.d
	picture = CULTURE_eventPicture

	hidden = yes

	mean_time_to_happen = {
		months = 240
	}

	trigger = {
		owner = {
			has_country_flag = abf_ukr_cossack_culture_spread
		}
		religion = orthodox
		NOT = {
			OR = {
				culture_group = east_slavic
			}
		}
		OR = {
			    area = crimea_area
				area = yedisan_area
				area = zaporizhia_area 
				area = azov_area
				area = lower_don_area
				area = circassia_area
				area = dagestan_area
				area = astrakhan_area
				area = saratov_area
				area = samara_area
				area = volga_area
				area = kazan_area
				area = lower_yik_area
				area = bashkiria_area
				area = galich_area
				area = perm_area
				area = kama_area
				area = nogai_area
	  }
	}

	option = {
		name = ab_new_cultures.23.a
		if = { 
			limit = {  #Black Sea 
				OR = {
				area = crimea_area
				area = yedisan_area
				area = zaporizhia_area 
				}
			}
			change_culture = aba_chernomoriya_cossack_culture
		}
		else_if = {
			limit = { #Don
				OR = {
				area = azov_area
				area = lower_don_area

				}
			}
			change_culture = aba_don_cossack_culture
		}
		else_if = { #Caucausia
			limit = {
				OR = {
				area = circassia_area
				area = dagestan_area
				}
			}
			change_culture = aba_terek_cossack_culture
		}
		else_if = { # Volga
			limit = {
				OR = {
				area = astrakhan_area
				area = saratov_area
				area = samara_area
				area = volga_area
				area = kazan_area
				area = tambov_area
				}
			}
			change_culture = aba_povolzyhe_cossack_culture
		}
		else_if = { # Yik
			limit = {
				OR = {
				area = lower_yik_area
				area = bashkiria_area
				area = galich_area
				area = perm_area
				area = kama_area
				area = nogai_area
				}
			}
			change_culture = aba_uralskiye_cossack_culture
		}
		else = {
			change_culture = ruthenian
		}
	}

}
