#Nikkei (Japanese) Spread
province_event = {
	id = ab_new_cultures.1
	title = ab_new_cultures.1.t
	desc = ab_new_cultures.1.d
	picture = CULTURE_eventPicture
	hidden = yes

	mean_time_to_happen = {
		months = 240
	}

#Will fire for all non japanese_g province in the philippines
trigger = {
	religion = shinto
    owner = {
        has_country_flag = nikkei_culture_spread
    }
    NOT = {
        culture_group = japanese_g
    }
    OR = {
        area = palawan_area
        area = luzon_area
        area = visayas_area
        area = west_mindanao_area
        area = mindanao_area
        area = southern_luzon_area
    }
}
	
	option = {
		name = ab_new_cultures.1.a	
		change_culture = japanese_nikkei
	}
}
